DROP DATABASE IF EXISTS anime;

CREATE DATABASE anime;



CREATE TABLE ANIME_T(
    Name VARCHAR(30) NOT NULL,    
    CONSTRAINT ANIME_PK PRIMARY KEY(Name)
);

CREATE TABLE USER_T(
    Username VARCHAR(30) NOT NULL,
    Password VARCHAR(30),
	
    CONSTRAINT USER_PK PRIMARY KEY(Username)
);


CREATE TABLE REVIEW_T(
	ReviewId int NOT NULL AUTO_INCREMENT,
    Username VARCHAR(30) NOT NULL,
    Anime VARCHAR(30) NOT NULL,
    Score INT NOT NULL,
    Review VARCHAR(200) NOT NULL,

    CONSTRAINT REVIEW_PK PRIMARY KEY(ReviewId),
    CONSTRAINT REVIEW_FK FOREIGN KEY(Username) REFERENCES USER_T(Username)
    /*,CONSTRAINT REVIEW_FK_2 FOREIGN KEY(Anime) REFERENCES ANIME_T(Name)*/
);
